{

Tanggal 	: 25 Oktober 2018
Program procedure untuk mengisi variable a larik dari 0 s.d n-1 menggunakan  looping for dan masukan nilai n dari pengguna.
}

Procedure isiArray ( a[ ]:n, n:int )

Deklarasi
	k : int

Algoritma
	for (k:=0 to n-1)
		output (“Masukkan nilai bilangan a[“k”]”)
		input (a[k])
	endfor

endProcedure
